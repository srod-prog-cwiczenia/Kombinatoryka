      n - liczba procesów;

      Dostepne[1..m] - ilość aktualnie dostępnych zasobów.\\
      Max[1..n,1..m] - maksymalna ilość z danego zasobu jaki życzy sobie dany proces.\\
      Pobrane[1..n,1..m] - ilość z danego zasobu jaką aktualnie pobrał dany proces.\\      
  \end{block}
  }
 
  \uncover<2->{
    \begin{block}{Mamy więc:}
     Potrzebne[1..n,1..m] := Max - Pobrane;
    \end{block}  
  }
  \end{frame}
